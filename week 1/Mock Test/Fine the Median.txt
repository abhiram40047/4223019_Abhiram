#include <stdio.h>
#include <stdlib.h>

int sortCompare(const void *p1, const void *p2) {
    return (*(int *)p1 - *(int *)p2);
}

int getMedian(int data[], int size) {
    qsort(data, size, sizeof(int), sortCompare);
    return data[size / 2];
}

int main() {
    int length;
    scanf("%d", &length);

    int values[length];
    for (int idx = 0; idx < length; idx++) {
        scanf("%d", &values[idx]);
    }

    int midVal = getMedian(values, length);
    printf("%d\n", midVal);

    return 0;
}
